package controllers

import (
	"github.com/astaxie/beego"
)

type TestArgController struct {
	beego.Controller
}

func (c *TestArgController) Get() {
	id := c.GetString("id")
	c.Ctx.WriteString(id)

	name := c.Input().Get("name")
	c.Ctx.WriteString(name)
}
